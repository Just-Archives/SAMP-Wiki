Description:

Elimina un veicolo precedentemente creato.


Parameters:
(vehicleid)
vehicleidID del veicolo da eliminare.


Return   Questa funzione non restituisce un valore specifico 

public OnPlayerCommandText(playerid, cmdtext[])
{
     //Prima lo script crea l'ID del veicolo e poi distrugge il veicolo con il comando /destroyveh
     if(strcmp(cmdtext, "/destroyveh", true) == 0)
     {
          new currentveh;
          currentveh = GetPlayerVehicleID(playerid);
          DestroyVehicle(currentveh);
          return 1;
     }
 
     return 0;
}
